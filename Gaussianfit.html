#!/usr/bin/env python3
"""
Fit a 1D Gaussian to the image of Vesta by collapsing the FITS file along the x–axis.

This does:
 1. Load “041755.fits” into a NumPy array - so load each file one by one
 2. Compute a 1D profile by summing the image along the y–axis (i.e., for each x, sum over all y).
 3. Fit that 1D profile with a Gaussian + constant background using Astropy’s modeling tools.
 4. Print the best‐fit parameters (amplitude, mean, σ, background) and show a plot of data vs. fit.

Requirements:
  • astropy
  • numpy
  • matplotlib


"""

import numpy as np
from astropy.io import fits
import matplotlib.pyplot as plt

from astropy.modeling import models, fitting


# If “041755.fits” is in the same folder as this script, you can leave this as-is.
# Otherwise, replace with the full path on your system.
filename = "./vestafits15may/041755.fits"


hdul = fits.open(filename)
data_full = hdul[0].data.astype(float)
hdul.close()

# If the image has more than 2 dimensions (e.g. data_full.shape = (1, ny, nx)),
# you can squeeze it down to 2D:
data_full = np.squeeze(data_full)


# We sum over y (axis=0) so that profile_x[i] = sum_y data_full[y, i].
profile_x = np.sum(data_full, axis=0)

# Create a corresponding x‐axis array of pixel indices
x = np.arange(profile_x.size)

# Fit a 1D Gaussian + constant background


# 4a. Make initial guesses for the Gaussian parameters:
#     • amplitude ≈ max(profile) − min(profile)
#     • mean ≈ x index of the maximum
#     • stddev ≈ some small value, e.g. 5 pixels
#     • background constant ≈ min(profile)
amp_init = np.max(profile_x) - np.min(profile_x)
mean_init = x[np.argmax(profile_x)]
std_init = 5.0
background_init = np.min(profile_x)

# 4b. Build a compound model: Gaussian1D + Const1D
g_init = models.Gaussian1D(amplitude=amp_init, mean=mean_init, stddev=std_init)
c_init = models.Const1D(amplitude=background_init)
model_init = g_init + c_init

# 4c. Use LevMarLSQFitter to fit the model to (x, profile_x)
fitter = fitting.LevMarLSQFitter()
fitted_model = fitter(model_init, x, profile_x)

# 4d. Extract best‐fit parameters
amplitude_fit = fitted_model.amplitude_0.value
mean_fit      = fitted_model.mean_0.value
stddev_fit    = fitted_model.stddev_0.value
background_fit= fitted_model.amplitude_1.value

print("Best‐fit 1D Gaussian parameters (fit to ∑_y intensity):")
print(f"  Amplitude  = {amplitude_fit:.2f}")
print(f"  Mean (x₀)  = {mean_fit:.2f} pixels")
print(f"  Stddev (σ) = {stddev_fit:.2f} pixels")
print(f"  Background = {background_fit:.2f} (constant offset)")


plt.figure(figsize=(8, 5))
plt.plot(x, profile_x, color="black", lw=1.2, label="Data (∑_y intensity)")
plt.plot(x, fitted_model(x), color="red", ls="--", lw=2, label="Gaussian + constant fit")

plt.xlabel("X pixel index")
plt.ylabel("Summed counts over all Y")
plt.title("1D Gaussian Fit to Vesta Image (summed along Y)")
plt.legend()
plt.grid(alpha=0.3)
plt.tight_layout()
plt.show()
